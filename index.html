function extractDomains(str) {

  const m = str.match(/https?:\/\/([a-z0-9.\-]+)/gi) || [];

  return [...new Set(m.map(u=>{ try{return new URL(u).hostname.toLowerCase();}catch(e){return null;} }).filter(Boolean))];

}

function isDomainAllowed(d) { return allowedDomains.some(a => d===a || d.endsWith('.'+a)); }

function isEncoded(str) {

  return /%[0-9A-Fa-f]{2}/.test(str) && !(/[<>]/.test(str) || (/&(?!amp;|lt;|gt;|quot;|apos;)/.test(str) && !/%26/.test(str)));

}

function looksLikeVAST(str) {

  return /<VAST/i.test(str) || /VASTAdTagURI/i.test(str) || /^https?:\/\/.+/i.test(str.trim());

}

function extractVersion(raw) {

  const m = raw.match(/VAST\s+version\s*=\s*["']?([\d.]+)["']?/i);

  return m ? m[1] : null;

}

function runOuterChecks(raw, ver) {

  const checks = [];

  checks.push({ name:'VAST Version 2.0', status: ver==='2.0'?'pass':'fail',

    detail: ver ? `Detected: VAST ${ver}` : 'No VAST version attribute found.',

    suggestion: ver!=='2.0' ? 'Request a VAST 2.0 tag. Other versions are rejected.' : null });

 

  const hasVapaid = /vapaid/i.test(raw);

  checks.push({ name:'VAPAID Attribute', status: hasVapaid?'fail':'pass',

    detail: hasVapaid?'VAPAID attribute detected.':'No VAPAID. ✓',

    suggestion: hasVapaid?'Remove VAPAID before trafficking.':null });

 

  const domains = extractDomains(raw);

  const blocked = domains.filter(d=>!isDomainAllowed(d));

  checks.push({ name:'Domain Allowlist', status: blocked.length?'fail':(domains.length?'pass':'warn'),

    detail: blocked.length ? `${blocked.length} unapproved domain(s): ${blocked.join(', ')}` : domains.length ? `All ${domains.length} domain(s) approved. ✓` : 'No external domains found.',

    suggestion: blocked.length ? `Remove URLs from: ${blocked.join(', ')}` : null });

 

  const hasImp = /<Impression/i.test(raw);

  checks.push({ name:'Impression Tracker', status: hasImp?'pass':'warn',

    detail: hasImp?'Impression URL present. ✓':'No <Impression> found.',

    suggestion: !hasImp?'Ask vendor to include an <Impression> URL.':null });

 

  const httpCount = (raw.match(/http:\/\//gi)||[]).length;

  checks.push({ name:'HTTPS Compliance', status: httpCount===0?'pass':'warn',

    detail: httpCount===0?'All URLs HTTPS. ✓':`${httpCount} non-secure HTTP URL(s).`,

    suggestion: httpCount>0?'Request vendor update all URLs to HTTPS.':null });

 

  const hasCT = /<ClickThrough/i.test(raw);

  checks.push({ name:'Click-Through URL', status: hasCT?'pass':'warn',

    detail: hasCT?'Click-through present. ✓':'No <ClickThrough> found.',

    suggestion: !hasCT?'Confirm if click destination is required.':null });

 

  const hasWrapper = /<Wrapper/i.test(raw)||/<VASTAdTagURI/i.test(raw);

  checks.push({ name:'Wrapper Detection', status: hasWrapper?'info':'pass',

    detail: hasWrapper?'Wrapper tag detected — see wrapper checks below.':'Inline VAST, no wrapper. ✓',

    suggestion: null });

 

  return checks;

}

 

function runWrapperChecks(raw) {

  const checks = [];

  const uriMatch = raw.match(/<VASTAdTagURI[^>]*>([\s\S]*?)<\/VASTAdTagURI>/i);

  const uriRaw   = uriMatch ? uriMatch[1].replace(/<!\[CDATA\[|\]\]>/g,'').trim() : null;

 

  checks.push({ name:'VASTAdTagURI Present', status: uriRaw?'pass':'fail',

    detail: uriRaw?`URI: ${uriRaw.substring(0,65)}…`:'No <VASTAdTagURI> found.',

    suggestion: !uriRaw?'Wrapper must contain a <VASTAdTagURI>.':null });

 

  if (uriRaw) {

    const enc = isEncoded(uriRaw);

    checks.push({ name:'Wrapper URI URL-Encoded', status: enc?'pass':'fail',

      detail: enc?'URI is correctly URL-encoded. ✓':'URI is NOT URL-encoded. Raw special chars detected.',

      suggestion: !enc?'Encode the URI: & → %26, = → %3D, ? → %3F. Ask vendor to encode before delivery.':null });

 

    const wDomains = extractDomains(uriRaw);

    const wBlocked = wDomains.filter(d=>!isDomainAllowed(d));

    checks.push({ name:'Wrapper URI Domain', status: wBlocked.length?'fail':(wDomains.length?'pass':'warn'),

      detail: wBlocked.length?`Unapproved domain(s): ${wBlocked.join(', ')}`:wDomains.length?`Approved: ${wDomains.join(', ')} ✓`:'No domain in wrapper URI.',

      suggestion: wBlocked.length?'Wrapper must redirect through approved domains only.':null });

  }

 

  const hasFallback = /fallbackOnNoAd\s*=\s*["']?true/i.test(raw);

  checks.push({ name:'fallbackOnNoAd', status: hasFallback?'pass':'warn',

    detail: hasFallback?'fallbackOnNoAd="true" set. ✓':'fallbackOnNoAd not set.',

    suggestion: !hasFallback?'Set fallbackOnNoAd="true" to handle empty ad responses gracefully.':null });

 

  return checks;

}

 

function runInnerTagChecks(raw) {

  const checks = [];

  const vm = raw.match(/VAST\s+version\s*=\s*["']?([\d.]+)["']?/i);

  checks.push({ name:'Inner — VAST 2.0', status: vm&&vm[1]==='2.0'?'pass':'fail',

    detail: vm?`Inner version: ${vm[1]}`:'No VAST version in inner tag.',

    suggestion: (!vm||vm[1]!=='2.0')?'Inner tag must also be VAST 2.0.':null });

 

  const isAnotherWrapper = /<Wrapper/i.test(raw)||/<VASTAdTagURI/i.test(raw);

  checks.push({ name:'Inner — Is Inline', status: isAnotherWrapper?'warn':'pass',

    detail: isAnotherWrapper?'Inner tag is also a wrapper (multi-level chain).':'Inner tag is inline. ✓',

    suggestion: isAnotherWrapper?'Request a flattened inline VAST. Nested wrappers increase timeout risk.':null });

 

  const hasMF = /<MediaFile/i.test(raw);

  checks.push({ name:'Inner — Video MediaFile', status: hasMF?'pass':'fail',

    detail: hasMF?'MediaFile detected. ✓':'No <MediaFile> — ad will not render.',

    suggestion: !hasMF?'Inner tag must have a <MediaFile> with a .mp4 URL.':null });

 

  const wm = raw.match(/width\s*=\s*["']?(\d+)["']?/i);

  const hm = raw.match(/height\s*=\s*["']?(\d+)["']?/i);

  const mW = wm?parseInt(wm[1]):null, mH = hm?parseInt(hm[1]):null;

  if (!mW&&!mH) {

    checks.push({ name:'Inner — Dimensions (1080×1920)', status:'warn',

      detail:'Cannot detect dimensions. Verify externally.',

      suggestion:'MediaFile must specify width="1080" height="1920".' });

  } else {

    const ok = mW===1080&&mH===1920;

    checks.push({ name:'Inner — Dimensions (1080×1920)', status: ok?'pass':'fail',

      detail: ok?`${mW}×${mH} ✓`:`Detected ${mW||'?'}×${mH||'?'}. Required 1080×1920.`,

      suggestion: !ok?'Request creative at exactly 1080×1920.':null });

  }

 

  const iBlocked = extractDomains(raw).filter(d=>!isDomainAllowed(d));

  checks.push({ name:'Inner — Domain Allowlist', status: iBlocked.length?'fail':'pass',

    detail: iBlocked.length?`Unapproved: ${iBlocked.join(', ')}`: 'All domains approved. ✓',

    suggestion: iBlocked.length?`Remove unapproved domains: ${iBlocked.join(', ')}`:null });

 

  const httpC = (raw.match(/http:\/\//gi)||[]).length;

  checks.push({ name:'Inner — HTTPS', status: httpC===0?'pass':'warn',

    detail: httpC===0?'All HTTPS. ✓':`${httpC} HTTP URL(s) in inner tag.`,

    suggestion: httpC>0?'All inner tag URLs must use HTTPS.':null });

 

  return checks;

}

 

// Quick summary for bulk table

function quickValidate(raw) {

  const ver = extractVersion(raw);

  const hasVAST = /<VAST/i.test(raw);

  if (hasVAST && ver !== '2.0') return { status:'reject', ver, score:0, issues:['VAST version rejected: '+ver], wrapper:false, dims:'N/A' };

 

  const outer   = runOuterChecks(raw, ver);

  const hasWrap = /<Wrapper/i.test(raw)||/<VASTAdTagURI/i.test(raw);

  const wrapChk = hasWrap ? runWrapperChecks(raw) : [];

  const all     = [...outer, ...wrapChk];

 

  const passed  = all.filter(c=>c.status==='pass').length;

  const failed  = all.filter(c=>c.status==='fail').length;

  const warned  = all.filter(c=>c.status==='warn').length;

  const score   = Math.round((passed/all.length)*100);

 

  // Dimension detection

  const wm = raw.match(/width\s*=\s*["']?(\d+)["']?/i);

  const hm = raw.match(/height\s*=\s*["']?(\d+)["']?/i);

  const mW = wm?parseInt(wm[1]):null, mH = hm?parseInt(hm[1]):null;

  const dims = (mW&&mH) ? `${mW}×${mH}` : '?';

 

  const issues = all.filter(c=>c.status==='fail'||c.status==='warn').map(c=>({n:c.name,s:c.status}));

  const overallStatus = failed>0?'fail':warned>0?'warn':'pass';

 

  return { status:overallStatus, ver:ver||'?', score, issues, wrapper:hasWrap, dims };

}

function handleFile(file) {

  if (!file) return;

  document.getElementById('dzFileName').textContent = '📎 ' + file.name;

  const ext = file.name.split('.').pop().toLowerCase();

  const reader = new FileReader();

 

  reader.onload = function(e) {

    let rows = [];

    if (ext === 'csv') {

      rows = parseCSV(e.target.result);

    } else {

      const wb = XLSX.read(e.target.result, {type:'array'});

      const ws = wb.Sheets[wb.SheetNames[0]];

      rows = XLSX.utils.sheet_to_json(ws, {header:1, defval:''});

    }

    if (rows.length < 2) { alert('File appears empty or could not be parsed.'); return; }

    processRows(rows);

  };

 

  if (ext === 'csv') reader.readAsText(file);

  else reader.readAsArrayBuffer(file);

}

 

function parseCSV(text) {

  const lines = text.split(/\r?\n/).filter(l=>l.trim());

  return lines.map(line => {

    const cells = [];

    let cur = '', inQ = false;

    for (let i=0; i<line.length; i++) {

      if (line[i]==='"') { inQ=!inQ; }

      else if (line[i]===','&&!inQ) { cells.push(cur.trim()); cur=''; }

      else cur+=line[i];

    }

    cells.push(cur.trim());

    return cells;

  });

}

 

function processRows(rows) {

  const headers = rows[0].map((h,i)=> h || `Column ${i+1}`);

  parsedColumns = headers.map((h,i)=>({

    header: h,

    values: rows.slice(1).map(r=> (r[i]||'').toString().trim())

  }));

 

  // Auto-detect VAST columns

  parsedColumns.forEach(col => {

    col.vastCount = col.values.filter(v => looksLikeVAST(v)).length;

    col.autoDetected = col.vastCount > 0;

  });

 

  // Render column picker

  const chips = document.getElementById('colChips');

  chips.innerHTML = parsedColumns.map((col,i)=>`

    <div class="col-chip ${col.autoDetected?'auto':''} ${selectedColIndex===i?'selected':''}"

         onclick="selectColumn(${i})"

         title="${col.vastCount} VAST tags detected">

      ${col.autoDetected?'✅ ':''}${col.header} ${col.autoDetected?`(${col.vastCount} tags)`:''}

    </div>`).join('');

 

  document.getElementById('colPicker').classList.add('visible');

 

  // Auto-select best column

  const best = parsedColumns.reduce((b,c,i)=> c.vastCount > (parsedColumns[b]||{vastCount:-1}).vastCount ? i : b, 0);

  if (parsedColumns[best] && parsedColumns[best].vastCount > 0) selectColumn(best);

}

 

function selectColumn(i) {

  selectedColIndex = i;

  document.querySelectorAll('.col-chip').forEach((c,idx)=> c.classList.toggle('selected', idx===i));

}

 

async function runBulkValidation() {

  let tags = [];

 

  if (bulkMode === 'paste') {

    tags = document.getElementById('bulkPasteInput').value

      .split('\n')

      .map(l=>l.trim())

      .filter(l=>l.length>0);

  } else {

    if (selectedColIndex === null) { alert('Please select the column containing VAST tags.'); return; }

    tags = parsedColumns[selectedColIndex].values.filter(v=>v.length>0);

  }

 

  if (tags.length === 0) { alert('No tags found to validate.'); return; }

 

  // Show progress

  const pw = document.getElementById('progressWrap');

  const pf = document.getElementById('progressFill');

  const pl = document.getElementById('progressLabel');

  pw.classList.add('visible');

  document.getElementById('bulkResults').style.display='none';

  bulkResultsData = [];

 

  for (let i=0; i<tags.length; i++) {

    await new Promise(r=>setTimeout(r,0)); // yield to UI

    const pct = Math.round(((i+1)/tags.length)*100);

    pf.style.width = pct+'%';

    pl.textContent = `Validating ${i+1} of ${tags.length} tags… (${pct}%)`;

    const result = quickValidate(tags[i]);

    bulkResultsData.push({ idx: i+1, raw: tags[i], ...result });

  }

 

  pw.classList.remove('visible');

  renderBulkTable(bulkResultsData);

  document.getElementById('exportBtn').style.display='inline-flex';

}

 

function renderBulkTable(data) {

  const total   = data.length;

  const rejected= data.filter(d=>d.status==='reject').length;

  const failed  = data.filter(d=>d.status==='fail').length;

  const warned  = data.filter(d=>d.status==='warn').length;

  const passed  = data.filter(d=>d.status==='pass').length;

 

  document.getElementById('bulkSummaryBar').innerHTML = [

    `<div class="summary-pill"><div class="dot" style="background:var(--muted)"></div>${total} Total</div>`,

    passed  ? `<div class="summary-pill"><div class="dot" style="background:var(--pass)"></div>${passed} Passed</div>`   : '',

    warned  ? `<div class="summary-pill"><div class="dot" style="background:var(--warn)"></div>${warned} Warnings</div>` : '',

    failed  ? `<div class="summary-pill"><div class="dot" style="background:var(--fail)"></div>${failed} Failed</div>`   : '',

    rejected? `<div class="summary-pill"><div class="dot" style="background:var(--fail)"></div>${rejected} Rejected</div>`: '',

  ].join('');

 

  renderTableRows(data);

  document.getElementById('bulkResults').style.display='block';

  document.getElementById('bulkResults').scrollIntoView({behavior:'smooth',block:'start'});

}

 

function renderTableRows(data) {

  const tbody = document.getElementById('bulkTableBody');

  tbody.innerHTML = data.map(d => {

    const preview = d.raw.replace(/</g,'&lt;').replace(/>/g,'&gt;').substring(0,80);

    const statusBadge = {

      pass:   `<span class="status-badge sb-pass">✓ Pass</span>`,

      fail:   `<span class="status-badge sb-fail">✕ Fail</span>`,

      warn:   `<span class="status-badge sb-warn">⚠ Warn</span>`,

      reject: `<span class="status-badge sb-reject">🚫 Rejected</span>`,

    }[d.status];

 

    const dimColor = d.dims==='1080×1920' ? 'var(--pass)' : d.dims==='?' ? 'var(--muted)' : 'var(--fail)';

 

    const issueHtml = d.status==='reject'

      ? `<div class="fail-item f">VAST ${d.ver} rejected — not 2.0</div>`

      : d.issues.length===0

        ? '<div class="fail-item" style="color:var(--pass)">No issues ✓</div>'

        : d.issues.slice(0,4).map(iss=>`<div class="fail-item ${iss.s==='fail'?'f':'w'}">

            ${iss.s==='fail'?'✕':'⚠'} ${iss.n}

          </div>`).join('') + (d.issues.length>4?`<div class="fail-item">+${d.issues.length-4} more…</div>`:'');

 

    return `<tr class="tr-${d.status}" data-status="${d.status}" data-raw="${encodeURIComponent(d.raw.substring(0,200))}">

      <td style="color:var(--muted);font-family:'Space Mono',monospace;">${d.idx}</td>

      <td><div class="tag-preview">${preview}</div></td>

      <td>${statusBadge}</td>

      <td class="score-cell" style="color:${d.score>=80?'var(--pass)':d.score>=50?'var(--warn)':'var(--fail)'}">${d.status==='reject'?'—':d.score+'%'}</td>

      <td style="font-family:'Space Mono',monospace;font-size:11px;color:${d.ver==='2.0'?'var(--pass)':'var(--fail)'}">${d.ver}</td>

      <td style="font-size:11px;">${d.wrapper?'<span style="color:var(--info)">⚠ Wrapper</span>':'<span style="color:var(--muted)">Inline</span>'}</td>

      <td style="font-family:'Space Mono',monospace;font-size:11px;color:${dimColor}">${d.dims}</td>

      <td><div class="fail-list">${issueHtml}</div></td>

    </tr>`;

  }).join('');

}

  }

  const search = document.getElementById('tableSearch').value.toLowerCase();

  if (search) filtered = filtered.filter(d=> d.raw.toLowerCase().includes(search) || d.dims.includes(search) || d.ver.includes(search));

function exportCSV() {

  if (!bulkResultsData.length) return;

  const headers = ['#','Tag Preview','Status','Score','VAST Version','Wrapper','Dimensions','Issues'];

  const rows = bulkResultsData.map(d=>[

    d.idx,

    '"'+d.raw.substring(0,120).replace(/"/g,"'")+'"',

    d.status,

    d.status==='reject'?'rejected':d.score+'%',

    d.ver,

    d.wrapper?'Yes':'No',

    d.dims,

    '"'+(d.status==='reject'?'VAST version rejected':d.issues.map(i=>i.n).join('; '))+'"'

  ]);

  const csv = [headers, ...rows].map(r=>r.join(',')).join('\n');

  const blob = new Blob([csv], {type:'text/csv'});

  const a = document.createElement('a'); a.href=URL.createObjectURL(blob); a.download='vast_validation_results.csv'; a.click();

}
